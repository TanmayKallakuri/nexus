CHANGE 1: ADD CONSTRUCT-PERCENTILE FEATURES
Priority: HIGHEST — do this first
============================================================

WHY:
  Right now the model sees person features and question embeddings
  separately. It has to DISCOVER that "this person is high on
  risk aversion" and "this question measures risk aversion" by
  itself. That's hard with 233 people.

  Adding this as a feature directly tells the model: "this person
  ranks 95th on what this question measures."

CHANGE 3: CALIBRATION LAYER (VARIANCE STRETCHING)
Priority: HIGH — do this after retraining with changes 1+2
============================================================

WHY:
  Even after changes 1 and 2, LightGBM will still somewhat
  regress to the mean. This post-hoc step stretches predictions
  to match the actual variance, without retraining.

HOW:

  After you get predictions from the ensemble, apply this
  PER QUESTION (not globally):

  def calibrate_predictions(test_df, stretch_factor=None):
      """
      Stretch predictions away from per-question mean
      to match actual variance.
      """
      calibrated = test_df.copy()

      for qid in calibrated['question_id'].unique():
          mask = calibrated['question_id'] == qid
          preds = calibrated.loc[mask, 'predicted'].values

          q_mean = preds.mean()
          q_pred_std = preds.std()

          if q_pred_std < 0.01:
              # No variance to stretch — skip
              continue

          if stretch_factor is None:
              # Auto-compute: target std ~1.1 (observed average across training)
              target_std = 1.1
              sf = target_std / q_pred_std
          else:
              sf = stretch_factor

          # Stretch
          stretched = q_mean + sf * (preds - q_mean)

          # Clamp to valid range
          num_opts = calibrated.loc[mask, 'num_options'].iloc[0]
          stretched = np.clip(stretched, 1, num_opts)

          calibrated.loc[mask, 'predicted'] = stretched

      return calibrated

  # Tune stretch_factor on validation:
  # Try values: 1.5, 2.0, 2.5, 3.0, 3.5
  # Pick the one that maximizes Pearson r on holdout questions

  best_r = -1
  best_sf = 1.0
  for sf in [1.5, 2.0, 2.5, 3.0, 3.5]:
      cal_df = calibrate_predictions(test_df, stretch_factor=sf)

      # Compute Pearson r on calibrated predictions
      correlations = []
      for qid in holdout_qids:
          q_mask = cal_df['question_id'] == qid
          actual = cal_df.loc[q_mask, 'answer_position'].values
          predicted = cal_df.loc[q_mask, 'predicted'].values
          if len(actual) >= 3 and np.std(actual) > 0 and np.std(predicted) > 0:
              from scipy.stats import pearsonr
              r, _ = pearsonr(actual, predicted)
              correlations.append(r)
      mean_r = np.mean(correlations)

      # Also compute accuracy
      cal_df['abs_error'] = np.abs(cal_df['predicted'] - cal_df['answer_position'])
      cal_df['range'] = cal_df['num_options'] - 1
      accuracy = 1 - (cal_df['abs_error'].mean() / cal_df['range'].mean())

      print(f"  stretch={sf}: accuracy={accuracy:.4f}, r={mean_r:.4f}")
      if mean_r > best_r:
          best_r = mean_r
          best_sf = sf

  print(f"\nBest stretch factor: {best_sf} (r={best_r:.4f})")

CHANGE 4: BUILD PERCENTILE BASELINE AND BLEND
Priority: MEDIUM — do this after changes 1-3
============================================================

WHY:
  A dead-simple model that maps person construct percentile
  directly to the answer scale has PERFECT variance by
  construction. Blending it with LightGBM corrects for
  variance collapse.

HOW:

  def percentile_baseline_predict(person_id, question_id, num_options,
                                   construct_mapping, person_construct):
      """
      Simple predictor: person percentile on matched construct → answer
      """
      # Find construct for this question
      cm = construct_mapping[construct_mapping['question_id'] == question_id]
      if cm.empty:
          return num_options / 2  # midpoint fallback

      construct_id = cm['construct_id'].iloc[0]

      # Find person's percentile on this construct
      pc = person_construct[
          (person_construct['person_id'] == person_id) &
          (person_construct['construct_id'] == construct_id)
      ]
      if pc.empty:
          return num_options / 2  # midpoint fallback

      percentile = pc['percentile'].iloc[0]

      # Map percentile to answer scale: 1 to num_options
      prediction = 1 + percentile * (num_options - 1)
      return prediction

  # Compute percentile baseline for all test rows
  test_df['percentile_pred'] = test_df.apply(
      lambda row: percentile_baseline_predict(
          row['person_id'], row['question_id'], row['num_options'],
          construct_mapping, person_construct
      ), axis=1
  )

  # Blend with LightGBM
  # Try different weights on validation
  best_r = -1
  best_w = 0.5
  for w_lgbm in [0.3, 0.4, 0.5, 0.6, 0.7, 0.8]:
      test_df['blended'] = w_lgbm * test_df['predicted'] + (1 - w_lgbm) * test_df['percentile_pred']

      correlations = []
      for qid in holdout_qids:
          q_mask = test_df['question_id'] == qid
          actual = test_df.loc[q_mask, 'answer_position'].values
          predicted = test_df.loc[q_mask, 'blended'].values
          if len(actual) >= 3 and np.std(actual) > 0 and np.std(predicted) > 0:
              r, _ = pearsonr(actual, predicted)
              correlations.append(r)

      mean_r = np.mean(correlations)
      print(f"  w_lgbm={w_lgbm}: r={mean_r:.4f}")
      if mean_r > best_r:
          best_r = mean_r
          best_w = w_lgbm

  print(f"\nBest blend: {best_w} LightGBM + {1-best_w} percentile (r={best_r:.4f})")

EXPECTED IMPACT:
  Correlation improves because percentile baseline has natural
  person-level spread that combats LightGBM's compression.
